
import java.util.Scanner;
class AddNumbers {
    int add(int a, int b) {
        return a + b;
    }
}
public class Q2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int y = sc.nextInt();
        AddNumbers obj = new AddNumbers();
        System.out.println(obj.add(x, y));
    }
}
